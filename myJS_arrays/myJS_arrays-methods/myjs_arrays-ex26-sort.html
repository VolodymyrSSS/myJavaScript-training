myjs-arrays: arrays_methods: exercises_Отсортировать пользователей по возрасту

Напишите функцию sortByAge(users), которая принимает массив объектов со свойством age и сортирует их по нему.

Например:
let oles = { name: 'Олесь', age: 25 };
let stepan = { name: 'Степан', age: 30 };
let olga = { name: 'Ольга', age: 28 };
let petro = { name: 'Петро', age: 31 };
let lesya = { name: 'Леся', age: 19 };

let arr = [ oles, stepan, olga, petro, lesya,];

sortByAge(arr);

// теперь [lesya, oles, olga, stepan, petro]
console.log( arr[0].name ) // lesya
console.log( arr[1].name ) // oles
console.log( arr[3].name ) // stepan


Решение:

Для решения задачи используем метод arr.sort(), который сортирует массив на месте, меняя в нём порядок элементов. Он возвращает отсортированный массив, но обычно возвращаемое значение игнорируется, так как изменяется сам arr. Или так - по умолчанию элементы сортируются как строки и если брать, как в задаче, за основу для сортирования возраст с числами, то '22' > '19' и на первом месте станет 'Олесь' а не 'Леся', а нам это не подходит.
Чтобы отсортировать массив, нам нужна функция, определяющая порядок, которая знает, как сравнивать его элементы - то есть так как нам нужно.


function sortByAge(arr) {
arr.sort((a, b) => a.age > b.age ? 1 : -1);
}
let oles = { name: 'Олесь', age: 22 };
let stepan = { name: 'Степан', age: 30 };
let olga = { name: 'Ольга', age: 28 };
let petro = { name: 'Петро', age: 31 };
let lesya = { name: 'Леся', age: 19 };

let arr = [ oles, stepan, olga, petro, lesya,];

sortByAge(arr);

// теперь [lesya, oles, olga, stepan, petro]
console.log( arr ) // console.log( arr[0].name ) // lesya
console.log( arr[1].name ) // oles
console.log( arr[3].name ) // stepan
