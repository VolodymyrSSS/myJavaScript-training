myjs-arrays: basic_exercises: подмассив наибольшей суммы

На входе массив чисел, например: arr = [1, -2, 3, 4, -9, 6].
Найти непрерывный подмассив в arr, сумма элементов в котором максимальна.
Функция getMaxSubSum(arr) должна возвращать эту сумму. Если все элементы отрицательные – ничего не берём(подмассив пустой) и сумма равна «0».

берём сумму выделенных элементов:
getMaxSubSum([-1, 2, 3, -9]) = 5
getMaxSubSum([-1, 2, 3, -9, 11]) = 11
getMaxSubSum([-2, -1, 1, 2]) = 3
getMaxSubSum([100, -9, 2, -3, 5]) = 100
getMaxSubSum([1, 2, 3]) = 6 здесь берём все
getMaxSubSum([-1, -2, -3]) = 0 здесь ничего


Решение:

Идём по массиву и накапливаем текущую частичную сумму элементов в переменной s.
Если s в какой-то момент становится отрицательной – присваиваем s=0.
Максимальный из всех s и будет ответом.


function getMaxSubSum(arr) {
let maxSum = 0;
let partialSum = 0;

for (let item of arr) { -для каждого элемента массива
partialSum += item; -добавляем значение элемента к partialSum
maxSum = Math.max(maxSum, partialSum); -запоминаем максимум на данный момент
if (partialSum < 0) partialSum = 0; -ноль если отрицательное
}

return maxSum;
}

// проверим работу
console.log( getMaxSubSum([-1, 2, 3, -9]) ); // 5
console.log( getMaxSubSum([-1, 2, 3, -9, 11]) ); // 11
console.log( getMaxSubSum([-2, -1, 1, 2]) ); // 3
console.log( getMaxSubSum([100, -9, 2, -3, 5]) ); // 100
console.log( getMaxSubSum([1, 2, 3]) ); // 6
console.log( getMaxSubSum([-1, -2, -3]) ); // 0


Этот алгоритм требует ровно 1 проход по массиву.
